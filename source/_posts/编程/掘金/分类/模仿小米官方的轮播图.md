        &#125;

    &#125;

    &#125;

    // 下一张
    function goNext() &#123;
        if (index < len - 1) &#123;
            index++
        &#125; else &#123;
            index = 0;
        &#125;
        goIndex()
    &#125;

    // 上一张
    function goPre() &#123;
        if (index === 0) &#123;
            index = len - 1
        &#125; else &#123;
            index--
        &#125;

    &#125;

    // 跨越式圆点
    for (var i = 0; i < points.length; i++) &#123;
        points[i].addEventListener('click', function () &#123;
            var pointIndex = this.getAttribute('data-index');
            index = pointIndex
            goIndex()
        &#125;)
    &#125;

    //节流
    //限制用户过分点击 带来不好体验

    // 下一张

        if (!lock) return

        &#125;, 1000);
    &#125;)

    // 上一张

        setTimeout(() => &#123;
            lock = true
        &#125;, 1000);
    &#125;)

    var timer;
    // 自动轮播

            goNext()
        &#125;, 1000);
    &#125;
