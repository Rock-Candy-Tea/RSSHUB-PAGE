<p><img class="content-image" src="https://www.zhihu.com/equation?tex=%5Ccos+x%3D%5Cfrac%7B1%7D%7B2%7D%5Cleft%28%5Cfrac%7B1%7D%7B%5Csqrt%5B3%5D%7B%5Ccos+3+x%2B%5Csqrt%7B%5Ccos%5E2+3+x-1%7D%7D%7D%2B%7B%5Csqrt%5B3%5D%7B%5Ccos+3+x%2B%5Csqrt%7B%5Ccos%5E2+3+x-1%7D%7D%7D%5Cright%29." alt referrerpolicy="no-referrer"></p>
